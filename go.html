time

package main
import (
    "fmt"
    "time"
)
func test() {
    start := time.Now()
    sum := 0
    for i := 0; i < 100000; i++ {
	sum++
    }
    elapsed := time.Since(start)
    fmt.Println("该函数执行完成耗时：", elapsed)
}

func main() {
    test()
}


该函数执行完成耗时： 38.669µs

sync

package main
import (
    "fmt"
    "time"
    "sync"
)
func main() {
    var wg sync.WaitGroup
    wg.Add(1)

    go func(){
	count("Sleep")
	wg.Done()
    }()
    wg.Wait()
}

func count(thing string) {
    for i := 1; i <= 5; i++ {
	fmt.Println(i, thing)
	time.Sleep(time.Millisecond*500)
    }

}
